SELECT ROWID ROW_ID ,BG_NUM,EVENT_DATE,TO_CHAR(TO_DATE(EVENT_DATE,'DD-MM-YYYY')+DENSE_RANK() OVER(PARTITION BY BG_NUM  ORDER BY ROWNUM )-1,'DD-MM-YYYY') NEW_EVENT_DATE FROM TF001 WHERE (BG_NUM,EVENT_DATE) IN(
